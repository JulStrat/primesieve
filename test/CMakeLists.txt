file(GLOB files "*.cpp")
foreach(file ${files})
    get_filename_component(binary_name ${file} NAME_WE)
    add_executable(${binary_name} ${file})
    target_link_libraries(${binary_name} libprimesieve)
    add_test(NAME ${binary_name} COMMAND ${binary_name})
endforeach()

if(BUILD_SHARED_LIBS OR WIN32)
    enable_language(C)
    file(GLOB files "*.c")
    foreach(file ${files})
        get_filename_component(binary_name ${file} NAME_WE)
        add_executable(${binary_name} ${file})
        target_link_libraries(${binary_name} libprimesieve)
        set_target_properties(${binary_name} PROPERTIES LINKER_LANGUAGE C)
        add_test(NAME ${binary_name} COMMAND ${binary_name})
    endforeach()
endif()
